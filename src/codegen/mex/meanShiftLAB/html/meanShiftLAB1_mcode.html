1function [newX] = meanShiftLAB(X,currX,bandwidth,activeIndicies)
 2    
 3    % loop for now because vectored is too big
 4    mG = (1/(2*pi))^(1/2);
 5    newX = currX;
 6    for i=1:size(activeIndicies,2)
 7        index = activeIndicies(i);
 8        D = sum((X-currX(:,index)).^2,1)./bandwidth(1);
 9
10        % then compute the kernel of D (including /bandwidth)
11        K = mG.*exp(-1/2.*(D));
12
13        % then compute the vectorized mean shift
14        newX(:,index) = sum(X*K',2)/sum(K);
15    end
16end
