1function [newX] = meanShiftLABXY(X,currX,bandwidth,activeIndicies)
 2    
 3    % loop for now because vectored is too big
 4    mG = (1/(2*pi))^(1/2);
 5    
 6    newX = currX;
 7    for i=1:size(activeIndicies,2)
 8        index = activeIndicies(i);
 9        D1 = sum((X(1:3,:)-currX(1:3,index)).^2,1)./bandwidth(1);
10        D2 = sum((X(4:5,:)-currX(4:5,index)).^2,1)./bandwidth(2);
11        % then compute the kernel of D (including /bandwidth)
12        K1 = mG.*exp(-1/2.*(D1));
13        K2 = mG.*exp(-1/2.*(D2));
14        K = K1.*K2;
15
16        % then compute the vectorized mean shift
17        newX(:,index) = sum(X*K',2)/sum(K);
18    end
19end
