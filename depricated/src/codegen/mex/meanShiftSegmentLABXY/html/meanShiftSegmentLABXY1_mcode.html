1function [pts_out] = meanShiftSegmentLABXY(pts,bandwidth,options)
 2coder.extrinsic('sprintfc');
 3% Compute the mean shift procedure on the input data and return the segmented data labels
 4%
 5% Adapted from:
 6% Comaniciu and Meer. "Mean Shift: A Robust Approach Toward Feature Space
 7% Analysis" IEE Transactions on Pattern Analysis and Machine Intelligence.
 8% Vol. 24, No. 5, May 2002.
 9    
10    % start by flattening to list of vectors
11    l = size(pts,1);
12    w = size(pts,2);
13    feature_size = size(pts,3);
14    X = reshape(pts,l*w,feature_size)';
15    currX = X;
16    
17    % then compute spatial list
18    [Xs,Ys] = meshgrid(1:l,1:w);
19    S = reshape(cat(3,Xs,Ys),l*w,2)';
20    currS = S;
21    
22    % loop until done shifting
23    activeIndicies = true(1,l*w);
24    mG = (1/(2*pi))^(1/2);
25    for i=1:options.maxLoops
26        
27        % use mean shift on the points
28        % loop for now because vectored is too big
29        newX = currX;
30        newS = currS;
31        for index=1:l*w
32            if activeIndicies(index)
33                D1 = sum((X(:,:)-currX(:,index)).^2,1)./bandwidth(1);
34                D2 = sum((S(:,:)-currS(:,index)).^2,1)./bandwidth(2);
35                % then compute the kernel of D (including /bandwidth)
36                K1 = mG.*exp(-1/2.*(D1));
37                K2 = mG.*exp(-1/2.*(D2));
38                K = K1.*K2;
39
40                % then compute the vectorized mean shift
41                newX(:,index) = sum(X*K',2)/sum(K);
42                newS(:,index) = sum(S*K',2)/sum(K);
43            end
44        end
45        
46        % see how many indicies are still active and update
47        delta = sum((currX-newX).^2,1) + sum((currS-newS).^2,1);
48        activeIndicies = delta > options.tolExit;
49        countActive = nnz(activeIndicies);
50        
51        % if debug display stats on iter
52        if options.debug
53            disp(['[',sprintfc('%d', i),'] countActive: ',sprintfc('%d', countActive)]);
54        end
55        
56        % update variables
57        currX = newX;
58        currS = newS;
59        
60        % exit if done
61        if(countActive == 0)
62            break;
63        end
64    end
65    pts_out = reshape(currX(1:3,:)',l,w,[]);
66end
